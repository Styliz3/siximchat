Getting Started

SimixChat API provides a simple yet powerful interface to access state-of-the-art language models. Here's how to get started:

JavaScript 
// Direct usage with fetch API - no library needed
async function callSimixChat(messages, model = 'llama-3-8b-8192') {
  const response = await fetch('https://simixchat.vercel.app', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      model: model,
      messages: messages,
      temperature: 0.7
    })
  });
  
  return await response.json();
}

Streaming Responses

Get real-time responses as they're generated for a better user experience:

JavaScript 
async function streamChatWithAI(onChunk) {
  const messages = [
    { role: 'system', content: 'You are a helpful assistant.' },
    { role: 'user', content: 'Write a short story about a robot.' }
  ];
  
  try {
    const response = await fetch('https://simixchat.vercel.app', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        model: 'llama-3.1-8b-instant',
        messages: messages,
        stream: true
      })
    });

    const reader = response.body.getReader();
    const decoder = new TextDecoder('utf-8');
    let buffer = '';
    let fullResponse = '';

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      
      const chunk = decoder.decode(value);
      buffer += chunk;
      
      const lines = buffer.split('\n');
      buffer = lines.pop() || '';  // Keep the last incomplete line in buffer
      
      for (const line of lines) {
        if (line.startsWith('data: ') && line !== 'data: [DONE]') {
          const data = JSON.parse(line.substring(6));
          const content = data.choices[0].delta.content || '';
          if (content) {
            fullResponse += content;
            onChunk(content);
          }
        }
      }
    }
    
    return fullResponse;
  } catch(error) {
    console.error('Error streaming from SimixChat API:', error);
  }
}

<script>
  document.addEventListener('DOMContentLoaded', function() {
    const chatMessages = document.getElementById('simixChatMessages');
    const userInput = document.getElementById('simixUserInput');
    const sendButton = document.getElementById('simixSendBtn');
    
    // Maintain conversation history
    let conversationHistory = [
      { role: 'system', content: 'You are a helpful assistant.' }
    ];

    async function streamChatResponse(message) {
      try {
        const response = await fetch('https://simixchat.vercel.app', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json'
          },
          body: JSON.stringify({
            model: 'llama-3.1-8b-instant', // Or any preferred model
            messages: conversationHistory,
            stream: true
          })
        });

        const reader = response.body.getReader();
        const decoder = new TextDecoder('utf-8');
        let buffer = '';
        let fullResponse = '';

        // Create typing indicator
        const typingIndicator = document.createElement('div');
        typingIndicator.className = 'simix-typing-indicator';
        for (let i = 0; i < 3; i++) {
          const dot = document.createElement('div');
          dot.className = 'simix-typing-dot';
          typingIndicator.appendChild(dot);
        }
        chatMessages.appendChild(typingIndicator);
        
        // Create bot message container
        const botMessageDiv = document.createElement('div');
        botMessageDiv.className = 'simix-message simix-bot-message';
        
        const messageContent = document.createElement('div');
        messageContent.className = 'simix-message-content';
        
        botMessageDiv.appendChild(messageContent);
        
        // Remove typing indicator after a brief delay
        setTimeout(() => {
          if (chatMessages.contains(typingIndicator)) {
            chatMessages.removeChild(typingIndicator);
            chatMessages.appendChild(botMessageDiv);
          }
        }, 1000);

        while (true) {
          const { done, value } = await reader.read();
          if (done) break;
          
          const chunk = decoder.decode(value);
          buffer += chunk;
          
          const lines = buffer.split('\n');
          buffer = lines.pop() || '';
          
          for (const line of lines) {
            if (line.startsWith('data: ') && line !== 'data: [DONE]') {
              try {
                const data = JSON.parse(line.substring(6));
                const content = data.choices[0].delta.content || '';
                if (content) {
                  fullResponse += content;
                  messageContent.textContent += content;
                  chatMessages.scrollTop = chatMessages.scrollHeight;
                }
              } catch (error) {
                console.error('Error parsing streaming data:', error);
              }
            }
          }
        }
        
        // Add the full response to conversation history
        conversationHistory.push({ role: 'assistant', content: fullResponse });
        
        return fullResponse;
      } catch (error) {
        console.error('Error:', error);
        
        // Add error message
        const errorDiv = document.createElement('div');
        errorDiv.className = 'simix-message simix-bot-message';
        const errorContent = document.createElement('div');
        errorContent.className = 'simix-message-content';
        errorContent.textContent = 'Sorry, there was an error. Please try again.';
        errorDiv.appendChild(errorContent);
        chatMessages.appendChild(errorDiv);
      }
    }

    function sendMessage() {
      const message = userInput.value.trim();
      if (message === '') return;
      
      // Add user message to chat
      const userMessageDiv = document.createElement('div');
      userMessageDiv.className = 'simix-message simix-user-message';
      const messageContent = document.createElement('div');
      messageContent.className = 'simix-message-content';
      messageContent.textContent = message;
      userMessageDiv.appendChild(messageContent);
      chatMessages.appendChild(userMessageDiv);
      
      // Clear input
      userInput.value = '';
      
      // Add to conversation history
      conversationHistory.push({ role: 'user', content: message });
      
      // Keep history limited to last 10 messages to avoid context length issues
      if (conversationHistory.length > 11) { // 1 system + 10 conversation turns
        conversationHistory = [
          conversationHistory[0], 
          ...conversationHistory.slice(conversationHistory.length - 10)
        ];
      }
      
      // Stream response
      streamChatResponse(message);
      
      // Scroll to bottom
      chatMessages.scrollTop = chatMessages.scrollHeight;
    }

    // Event listeners
    sendButton.addEventListener('click', sendMessage);
    userInput.addEventListener('keypress', (e) => {
      if (e.key === 'Enter') {
        sendMessage();
      }
    });
  });
</script>
